SELECT SUM(pageviews) as totalPageviews FROM Total;

SELECT SUM(pageviews) as SessaoUsuarios FROM Total
GROUP BY idVisitante;

SELECT date, SUM(visits) as totalSessao FROM Total
GROUP BY date;

SELECT date, (SUM(timeOnSite)/SUM(visits)) as MediaSessao FROM Total
GROUP BY date;

SELECT SUM(pageviews) as totalGeral FROM Total
GROUP BY browser;
